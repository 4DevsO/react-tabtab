Advanced Tabs

(check the console to see the click response)

With these callback function, you can combine them with your data to make the add tab, delete tab action.


var App = React.createClass({
  render: function() {
    return (
      <Tabs activeKey={2} 
            style="tabtab__folder__" 
            addTab={true}
            handleAddTab={handleAddTab}
            tabDeleteButton={true}
            handleTabDeleteButton={handleTabDeleteButton}
            deleteAllButton={true}
            handleDeleteAllButton={handleDeleteButton}
            handleTabClick={handleTabClick}>
        <Panel title="Tab1">
          Content1
        </Panel>
        <Panel title="Tab2">
          Content2
        </Panel>
        <Panel title="Tab3">
          Content3
        </Panel>
      </Tabs>
    )
  }
})
